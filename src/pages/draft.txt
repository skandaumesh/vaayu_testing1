  const [current, setCurrent] = useState(0);

  useEffect(() => {
    const timer = setInterval(() => {
      setCurrent((prev) => (prev + 1) % sliderContent.length);
    }, 50000); // ⏱️ 50 seconds
    return () => clearInterval(timer);
  }, []);

  const handlePrev = () => {
    setCurrent((prev) => (prev === 0 ? sliderContent.length - 1 : prev - 1));
  };

  const handleNext = () => {
    setCurrent((prev) => (prev + 1) % sliderContent.length);
  };

  return (
    <div>
      {/* ✅ Hero Section with Simple Arrows and Dots */}
      <div
        className="relative w-full h-[90vh] bg-cover bg-center bg-no-repeat flex flex-col justify-center px-5 lg:px-32 text-white transition-all duration-700"

        {/* Dark Overlay */}
        <div className="absolute inset-0 bg-black opacity-30 z-0" />

<div className="z-10 pt-24 md:pt-0">

            {sliderContent[current].title}
          </h1>
          <p className="mt-4 text-lg max-w-2xl">
            {sliderContent[current].text}
          </p>
        </div>
{/* ← Left Arrow */}
<button
  onClick={handlePrev}
  className="absolute left-1 sm:left-4 top-[55%] sm:top-1/2 transform -translate-y-1/2 text-white text-2xl z-20"
>
  &#10094;
</button>

{/* → Right Arrow */}
<button
  onClick={handleNext}
  className="absolute right-1 sm:right-4 top-[55%] sm:top-1/2 transform -translate-y-1/2 text-white text-2xl z-20"
>
  &#10095;
</button>



        {/* ● Dots */}
        <div className="absolute bottom-6 left-1/2 transform -translate-x-1/2 flex space-x-2 z-20">
          {sliderContent.map((_, index) => (
            <button
              key={index}
              onClick={() => setCurrent(index)}
              className={`w-2 h-2 rounded-full ${
                index === current ? "bg-white" : "bg-gray-400"
              } hover:bg-white transition`}
            ></button>
          ))}
        </div>
      </div>

      {/* Stats Cards */}
      <StatsCards />
